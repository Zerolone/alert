带回调的alert
//带回调函数的alert调用方式
$.alert('这是alert',function(){
	$.alert('你点击了确定')
})
带回调的confirm
//带回调函数的confirm调用方式
$.confirm('这是confirm',function(a){
	$.alert('你点击了'+(a?'确定':'取消'))
})
return false阻止对话框关闭
//回调函数中的this指向对话框对象
//return false可以阻止对话框关闭
$.confirm('点击确定不会关闭对话框！！', function(e) {
	if(e){
		this.content('回调函数阻止了对话框关闭，点击取消才能关闭！！');
		return false;
	}
})
显示一条tips消息
自定义按钮文字
//自定义按钮文字
var dialog = $.confirm('自定义按钮文字');
dialog.ok('ok').cancel('cancel');
自定义外观尺寸
//自定义宽度，自定义内容区域的padding宽度
$.confirm('<iframe src="https://m.baidu.com/" width="100%" height="440" frameborder="no"/></iframe>').padding(0).width(320).ok('关闭');

使用文档：

版本API说明
PC/mob $.alert(str[,fn]) 弹出一个只包含确认按钮的对话框
PC/mob $.confrim(str[,fn]) 弹出一个包含确认和取消按钮的对话框
PC $.tips(str[,time]) 弹出一个自动消失的悬浮提示
mob $.tips(str) 弹出一个自动消失的悬浮提示，不支持自定义时间
版本方法说明 

var dialog = $.alert('下面方法API中的dialog对象是这么获得的')
$.confirm('回调中的this也是dialog对象',function(e){
  //这里的this也是dialog对象
  e||this.content('这样可以改变中间的内容')
  return e;
})
PC/mob dialog.content(str) 修改对话框的内容
PC/mob dialog.ok(str) 修改确认按钮文字
PC/mob dialog.cancel(str) 修改取消按钮文字
PC/mob dialog.close() 主动关闭对话框
PC dialog.padding(num) 设置对话框消息容器的padding值
PC dialog.width(num) 设置对话框总宽度
